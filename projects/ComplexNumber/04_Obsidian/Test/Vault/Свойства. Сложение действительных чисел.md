var scene = new MathSceneCanvas("canvas1");
var a1 = scene.AddSingle(new Complex(0.8, 0), "#FF0000", SingleMathObjectType.Real, 2, new Vector2(100, 100), new Vector2(200, 200));
var a2 = scene.AddSingle(new Complex(0.5, 0), "#0000FF", SingleMathObjectType.Real, 2, new Vector2(400, 100), new Vector2(200, 200));
scene.AddSum(a1, a2, "#00FF00", SumMathObjectType.Front, true, 2, new Vector2(700, 100), new Vector2(200, 200));
scene.Repaint();
