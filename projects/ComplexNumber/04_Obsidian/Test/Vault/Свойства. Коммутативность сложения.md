```js 
var scene = new MathSceneCanvas("canvas1");
var a1 = scene.AddSingle(new Complex(0.8, 1), "#FF0000", SingleMathObjectType.Simple, 2, new Vector2(100, 100), new Vector2(200, 200));
var a2 = scene.AddSingle(new Complex(0.5, -0.8), "#0000FF", SingleMathObjectType.Simple, 2, new Vector2(400, 100), new Vector2(200, 200));
var s1 = scene.AddSum(a1, a2, "#00FF00", SumMathObjectType.Front, false, 2, new Vector2(100, 400), new Vector2(200, 200));
var s2 = scene.AddSum(a1, a2, "#00FF00", SumMathObjectType.Back, false, 2, new Vector2(400, 400), new Vector2(200, 200));
scene.Repaint();

a1.SetText("x");
a2.SetText("y");
s1.SetText("x + y");
s2.SetText("y + x");
```
